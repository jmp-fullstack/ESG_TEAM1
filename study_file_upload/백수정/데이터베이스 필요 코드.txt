중요 코드
- 확인
SHOW DATABASES; #데이터 베이스 확인
SHOW TABLES; #테이블 확인

- 접근
명령 프롬포트에서 실행시 : mysql 경로까지 접근후 mysql -u {root} -p
USE {your_base} #데이터베이스 들어가기

- 만들기
CREATE DATABASE {DB_NAME} CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;   #유니코드 문자로 만듬
CREATE TABLE news ( id INT NOT NULL AUTO_INCREMENT,    # null값 허용안함 주 키 지정할꺼임(중복안됨, 무결성)
	date DATE, # DATE유형-날짜형식으로 저장한다는 뜻
	title VARCHAR(255), # 제목 255자
	content TEXT, # 매우 긴 텍스트까지 저장
	label VARCHAR(16), # 짧은 문자열
	PRIMARY KEY(id)); # 주키 설정

------------------------------------vscode에서 실행할때---------------------------------------
import mysql.connector

conn = pymysql.connect(
    host='localhost', # 어떤 컴퓨터에 있는 데이터베이스에 접속할 것인지?
    # 사용자 계정
    user='esg_project',
    # 비밀번호
    password = open('{파일경로}').read().strip(), # secret 파일 열어서, 읽은 뒤, 앞 뒤 공백 삭제
    database='esg_project', # 사용할 데이터베이스 선택 
    charset='utf8mb4'
)

# 데이터베이스 커서 생성
cursor = conn.cursor()  # with conn.cursor() as cursor 가능

# 삽입할 데이터
name = "John"
age = 30
city = "New York"

# INSERT 쿼리 작성
sql = "INSERT INTO users (name, age, city) VALUES (%s, %s, %s)"
# 쿼리 실행
cursor.execute(sql, (name, age, city))

# 변경 사항을 커밋하여 데이터베이스에 반영
conn.commit()

# 연결 및 커서 닫기 # with as 사용시 안해도됨
cursor.close()
